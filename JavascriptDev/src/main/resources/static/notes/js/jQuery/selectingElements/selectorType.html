Select all element of the same type

Open console.log() to see selector for all <p></p>

var arrayOfPs = $("p");

var arrayOfPsInnerHTML = [];
$.each(arrayOfPs, function(arrayIndex, value) {
    console.log("arrayIndex: ", arrayIndex, "value: ", value);
    arrayOfPsInnerHTML.push(value);
});
